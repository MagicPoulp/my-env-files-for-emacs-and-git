# Existing USB stick

To modify an existing ISO on a USB stick, one just needs to mount and to add 2 files
/preseed.cfg
/boot/grub/grub.cf

sudo mkdir -p /mnt/usb_boot
sudo fdisk -l
sudo mount /dev/sda1 /mnt/usb_boot

# New USB stick

The USB stick must be formated FAT32, with a GPT table, and with the boot esp flag

not sda1, but sda in below
sudo fdisk -l
sudo dd if=/path/to/your/image.iso of=/dev/sdX bs=4M status=progress
sudo sync

# to create a .ISO to be be used for installing a VM or for flashing a USB sticks, here is how

sudo mkdir -p /mnt/debian_iso
sudo mount -o loop /home/user/Downloads/debian-13.3.0-amd64-netinst.iso /mnt/debian_iso/
sudo mkdir /opt/iso_workspace
sudo rsync /mnt/debian_iso/ /opt/iso_workspace
sudo rsync -ahPHAXx --delete /mnt/debian_iso/ /opt/iso_workspace

cd /opt/iso_workspace/
cp ~/Downloads/grub.cfg boot/grub/grub.cfg
# set the right language for the keyboard in the preseed file
# Note we have put 512MB for /boot because 300MB was not enough to install a second kernel
sudo cp ~/Downloads/preseed.cfg .

sudo apt update
sudo apt install isolinux syslinux-common xorriso

xorriso -as mkisofs -r -V "Modified Debian" \
  -J -joliet-long -cache-inodes \
  -isohybrid-mbr /usr/lib/ISOLINUX/isohdpfx.bin \
  -b isolinux/isolinux.bin -c isolinux/boot.cat \
  -no-emul-boot -boot-load-size 4 -boot-info-table \
  -eltorito-alt-boot -e boot/grub/efi.img -no-emul-boot \
  -isohybrid-gpt-basdat \
  -o /home/user/Downloads/custom-debian-13.3.0-amd64-netinst.iso /opt/iso_workspace

sudo usermod -aG libvirt-qemu $USER
sudo usermod -aG libvirt $USER
sudo usermod -aG kvm $USER
#logout and relogin
sudo cp /home/user/Downloads/custom-debian-13.3.0-amd64-netinst.iso /var/lib/libvirt/images/
sudo chown libvirt-qemu:libvirt-qemu -R /var/lib/libvirt/images

# 1. Start the network right now
sudo virsh net-start default

# 2. Make sure it starts automatically on every reboot
sudo virsh net-autostart default
sudo virsh net-list --all

virsh vol-list images
virsh pool-refresh images

# the ///system is mandatory to have a performance network
virt-install \
  --connect qemu:///system \
  --name debian13-vm \
  --boot uefi \
  --ram 1024 \
  --vcpus 2 \
  --disk path=/var/lib/libvirt/images/debian-vm.qcow2,size=20,bus=virtio \
  --os-variant debian13 \
  --network network=default,model=virtio \
  --graphics vnc \
  --cdrom /var/lib/libvirt/images/custom-debian-13.3.0-amd64-netinst.iso \
  --noautoconsole
